Usage Example (TypeScript/JavaScript)

// Import the package
import { CircleAnimations } from 'realm-loader-npm'

// Create animations
const radialPulse = CircleAnimations.createRadialPulse('#radial-pulse .circle-container', {
  color: '#ff6b6b',
  speed: 1.5,
  width: 180,
  height: 180
})

const orbitalPulse = CircleAnimations.createOrbitalPulse('#orbital-pulse .circle-container', {
  color: '#4ecdc4',
  speed: 1.0
})

const pendulumWave = CircleAnimations.createPendulumWave('#pendulum-wave .circle-container', {
  color: '#45b7d1',
  speed: 0.8
})

// Control animations
radialPulse.start()
orbitalPulse.start()
pendulumWave.start()

// Update configuration
radialPulse.updateConfig({ color: '#ff0000', speed: 2.0 })

// Stop animations
radialPulse.stop()
orbitalPulse.destroy()
