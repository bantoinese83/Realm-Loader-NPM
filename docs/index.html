🚀 Quick Start

Installation

npm install realm-loader-npm

Basic Usage

import { CircleAnimations } from 'realm-loader-npm' const animation = new CircleAnimations({ container: '#my-container', animation: 'radial-pulse', config: { color: '#4ecdc4', speed: 1.5, width: 200, height: 200 } }) animation.start()

React Usage

import { RadialPulse } from 'realm-loader-npm/react' function App() { return ( <RadialPulse preset="loading-fast" theme="ocean" onStart={() => console.log('Started!')} /> ) }

Vue Usage

<template> <RadialPulse :preset="'loading-fast'" :theme="'ocean'" @start="onStart" /> </template> <script setup> import { RadialPulse } from 'realm-loader-npm/vue' </script>
