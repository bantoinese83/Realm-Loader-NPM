  config={{ size: 100, theme: "cyberpunk" }}

// Basic JavaScript usage

  container: '#my-container',
  animation: 'spiral-galaxy',
  config: {
    size: 100,
    theme: 'cyberpunk'
  },
  autoStart: true
})
